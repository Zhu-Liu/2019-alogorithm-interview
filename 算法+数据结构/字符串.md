## 翻转字符串里的单词
> LeetCode/[翻转字符串里的单词](https://leetcode-cn.com/problems/reverse-words-in-a-string/)
```c++
class Solution {
public:
    void reverseWords(string &s) {
        istringstream is(s);
        string temp;
        is>>s; //此时s为字符串中第一个单词
        while(is>>temp) s =temp +" " +s;
        if(!s.empty() && s[0] == ' ') s = "";
    }
};
```

```Python
class Solution(object):
    def reverseWords(self, s):
        """
        :type s: str
        :rtype: str
        """
        s = s.strip() ## 去掉字符串首尾的所有空格
        s = s.split() ## 按任意连续的空格划分字符串，得到列表s
        s.reverse() ## 翻转列表s
        s = ' '.join(s) ##用一个空格把列表s连接成字符串 
        return s
```

class Solution {
public:
    string reverseString(string s) {
        string tempS="";
            
        for(int i=s.size()-1;i>=0;i--)
            tempS+=s[i];
        return tempS;
    }
};

## 字符串转换成整数
> LeetCode/[字符串转换成整数](https://leetcode-cn.com/problems/string-to-integer-atoi/)

```c++
class Solution {
public:
    int myAtoi(string str) {
        int sign = 1;
        int i = 0;
        int result = 0;
        int digit = 0;
        while(str[i]==' ')i++; // 去除空格
        if(i<str.size() && str[i]=='-'){sign=-1; i++;} // 记录符号
        else if(i<str.size() && str[i]=='+'){sign=1; i++;}
        while(i<str.size())
        {
            if(isdigit(str[i]))
            {
                digit = str[i] - '0';
                if(sign*result > (INT_MAX - digit)/10) return INT_MAX; // 如果result已经大于最大整数
                if(sign*result < (INT_MIN + digit)/10) return INT_MIN; // 如果result已经小于最小整数
                result = 10*result + str[i] - '0';
            }
            else 
                return sign*result;
            i++;
        }
    return sign*result;
    }
};
```

## 括号生成
> LeetCode/[括号生成](https://leetcode-cn.com/problems/generate-parentheses/submissions/)   

**描述**
```
给出 n 代表生成括号的对数，请你写出一个函数，使其能够生成所有可能的并且有效的括号组合。

例如，给出 n = 3，生成结果为：

[
  "((()))",
  "(()())",
  "(())()",
  "()(())",
  "()()()"
]
```
- 回溯法
```c++
class Solution {
    vector<string> res;
public:
    vector<string> generateParenthesis(int n) {
        string s = "(";
        dfs(n-1, n, s);
        return res;
    }
    
    void dfs(int l, int r, string s){
        if(l == 0 && r == 0){
            res.push_back(s);
            return;
        }
        if(r > 0 && r > l){
            dfs(l, r-1, s+")"); // trick:不用中间值保留参数
        }
        if(l > 0){
            dfs(l-1, r, s+"(");
        }

    }
};
```

## 使括号有效的最少添加
> LeetCode/[使括号有效的最少添加](https://leetcode-cn.com/problems/minimum-add-to-make-parentheses-valid/)

**描述**
```
给定一个由 '(' 和 ')' 括号组成的字符串 S，我们需要添加最少的括号（ '(' 或是 ')'，可以在任何位置），以使得到的括号字符串有效。

从形式上讲，只有满足下面几点之一，括号字符串才是有效的：

它是一个空字符串，或者
它可以被写成 AB （A 与 B 连接）, 其中 A 和 B 都是有效字符串，或者
它可以被写作 (A)，其中 A 是有效字符串。
给定一个括号字符串，返回为使结果字符串有效而必须添加的最少括号数。
```
- 用栈，实际可以用一个变量来解决
```c++
class Solution {
public:
    int minAddToMakeValid(string S) {
        stack<char> s;
        int res = 0;
        for(auto i : S){
            if(i == '(')
                s.push(i);
            else
            {
                if(!s.empty() && s.top() == '(')
                    s.pop();
                else
                    res += 1;
            }
        }
        return res + s.size();
    }
};
```

## 删除无效的括号  
> LeetCode/[删除无效的括号](https://leetcode-cn.com/problems/remove-invalid-parentheses/)  
> Bilibili/[花花](https://www.bilibili.com/video/av31665179)

**描述**
```
删除最小数量的无效括号，使得输入的字符串有效，返回所有可能的结果。

说明: 输入可能包含了除 ( 和 ) 以外的字符。

示例 1:

输入: "()())()"
输出: ["()()()", "(())()"]
示例 2:

输入: "(a)())()"
输出: ["(a)()()", "(a())()"]
示例 3:

输入: ")("
输出: [""]
```

```c++
class Solution {
    vector<string> res;
public:
    vector<string> removeInvalidParentheses(string s) {
        int l = 0;
        int r = 0;
        for(auto i : s){
            if(i == '(')
                l++;
            else if(i == ')'){
                if(l > 0)
                    l--;
                else
                    r++;
            }
        }
        
        dfs(l, r, 0, s);
        return res;
    }
    
    bool isValid(string& s){
        int count = 0;
        for(auto i : s){
            if(i == '(')
                count++;
            if(i == ')')
                count--;
            if(count < 0)
                return false;
        }
       return count == 0;
    }
    
    void dfs(int l, int r, int start, string s){
        if(l == 0 && r == 0){
            if(isValid(s))
                res.push_back(s);
            return; // 注意return
        }
        
        for(int i = start; i < s.size(); i++){
            if(i != start && s[i] == s[i - 1])  // 防止重复搜索
                continue;
         
            string curr = s;
            curr.erase(i, 1); // 注意要使用中间变量
            if(l > 0 && s[i] == '(')
                dfs(l - 1, r, i, curr);
            if(r > 0 && s[i] == ')')
                dfs(l, r - 1, i, curr);
           
        }
        return;
    }
};
```

## 最短完整词
> LeetCode/[最短完整词](https://leetcode-cn.com/problems/shortest-completing-word/submissions/)
> bilibili/[花花酱]/(https://www.bilibili.com/video/av31665016)

**描述**
```
输入：licensePlate = "1s3 PSt", words = ["step", "steps", "stripe", "stepple"]
输出："steps"
说明：最短完整词应该包括 "s"、"p"、"s" 以及 "t"。对于 "step" 它只包含一个 "s" 所以它不符合条件。同时在匹配过程中我们忽略牌照中的大小写。
```

```c++
class Solution {
    bool match(const string &s, const vector<int>& dic){
        vector<int> dic_s(26, 0);
        
        for(char c : s)
            dic_s[c - 'a']++;
        
        for(int i = 0; i < 26; i++){
            if(dic_s[i] < dic[i])
                return false;
        }
        return true;
    }
public:
    string shortestCompletingWord(string licensePlate, vector<string>& words) {
        vector<int> plateDictionary(26, 0);
        
        for(char c : licensePlate){
            if(isalpha(c))
                plateDictionary[tolower(c) - 'a']++;
        }
        
        int ans = INT_MAX;
        string ans_string;
        for(string s : words){
            if(s.size() >= ans)
                continue;
            if(!match(s, plateDictionary))
                continue;
            ans = s.size();
            ans_string  = s;
        }
        return ans_string;
    }
};
```

## 面试题：自己实现strcpy()
```c++
char* mystrcpy(char* source, char* dest){
	assert(source != nullptr);
	assert(dest != nullptr); // 判空

	int size = strlen(source) + 1;// 别忘了数组最后的'\0'
	if(dest < source || source + size < dest){
		char* s = source;
		char* d = dest;
		while(size--)
			*s++ = *d++;
	}
	else{ // 内存重叠
		char* d = dest + size - 1;
		char* s = source + size - 1; 
		while(size--)
			*d-- = *s--;
	}
	return dest; // 别忘了返回
}
```
