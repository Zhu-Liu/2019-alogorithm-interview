## 岛屿的个数

**描述**
```
给定一个由 '1'（陆地）和 '0'（水）组成的的二维网格，计算岛屿的数量。一个岛被水包围，并且它是通过水平方向或垂直方向上相邻的陆地连接而成的。你可以假设网格的四个边均被水包围。

示例 1:

输入:
11110
11010
11000
00000

输出: 1
```
- dfs
```c++
class Solution {
    int n,m;
public:
    int numIslands(vector<vector<char>>& grid) {
        if(grid.empty() || grid[0].empty()) return 0;
        n = grid.size();
        m = grid[0].size();
        int ret = 0;
        for(int i = 0; i < n; i++)
            for(int j = 0; j < m; j++)
            {
                if(grid[i][j] == '1')
                {
                    ret += 1;
                    dfs(grid, i, j);
                }
            }
        
        return ret;
    }
    
    void dfs(vector<vector<char>>& grid, int i, int j)
    {
        if((i<0) || (i>=n) || (j<0) || (j>=m)) return;
        
        if(grid[i][j] == '1')
        {
            grid[i][j] = '0';
            dfs(grid, i+1, j);
            dfs(grid, i, j+1);
            dfs(grid, i-1, j);
            dfs(grid, i, j-1);
        }
        else
            return;
    }
};
```

- bfs
- 注意在入队前要置0，否则会重复入队导致超时
```c++
class Solution {
    int n,m;
public:
    int numIslands(vector<vector<char>>& grid) {
        if(grid.empty() || grid[0].empty()) return 0;
        n = grid.size();
        m = grid[0].size();
        int ret = 0;
        for(int i = 0; i < n; i++)
            for(int j = 0; j < m; j++)
            {
                if(grid[i][j] == '1')
                {
                    ret += 1;
                    bfs(grid, i, j);
                }
            }
        
        return ret;
    }
    
    void bfs(vector<vector<char>>& grid, int i, int j)
    {
        queue<pair<int, int>> q;
        
        q.push({i,j});
        grid[i][j] = '0';
        while(!q.empty())
        {
            int r = q.front().first;
            int c = q.front().second;
            q.pop();
     
            if((r > 0) && (grid[r-1][c] == '1'))   
            {
                q.push({r-1, c});
                grid[r-1][c] = '0';
            }
           if((r < (n - 1)) && (grid[r + 1][c] == '1'))
            {
                q.push({r+1, c});
                grid[r+1][c] = '0';
            }
            if((c > 0) && (grid[r][c-1] == '1'))
            {
                q.push({r, c-1});
                grid[r][c-1] = '0';
            }  
            if((c < (m - 1)) && (grid[r][c+1] == '1'))
            {
                q.push({r, c+1});
                grid[r][c+1] = '0';
            }
        }

    }
};
```

## 字典序排数
> LeetCode/[字典序排数](https://leetcode-cn.com/problems/lexicographical-numbers/submissions/)

**描述**
```
给定一个整数 n, 返回从 1 到 n 的字典顺序
```
- 通过dfs暴搜找到字典序
```c++
class Solution {
    vector<int> res;
public:
    vector<int> lexicalOrder(int n) {
        for(int i = 1; i <= 9; i++)
            dfs(n, i);
        return res;
    }
    
    void dfs(int n, int i){
        if(i > n)
            return;
        res.push_back(i);
        int t;
        for(int j = 0; j <= 9; j++){
            t = i * 10 + j;
            if(t > n)
                return;
            else
                dfs(n, t);
        }
    }
};
```


## 课程表
> LeetCode/[课程表](https://leetcode-cn.com/problems/course-schedule/)
> bilibili/[花花酱](https://www.bilibili.com/video/av38557948?t=739)

**描述**
```
现在你总共有 n 门课需要选，记为 0 到 n-1。

在选修某些课程之前需要一些先修课程。 例如，想要学习课程 0 ，你需要先完成课程 1 ，我们用一个匹配来表示他们: [0,1]

给定课程总量以及它们的先决条件，判断是否可能完成所有课程的学习？

示例 1:

输入: 2, [[1,0]] 
输出: true
解释: 总共有 2 门课程。学习课程 1 之前，你需要完成课程 0。所以这是可能的。
```

- 拓扑排序
```c++
class Solution {
public:
    bool canFinish(int numCourses, vector<pair<int, int>>& prerequisites) {
        vector<int> flag(numCourses);
        vector<vector<int>> Map(numCourses);  
        /* 0::unvisted; 1::visting; 2::visted */
        
        for(auto &i : prerequisites){
            Map[i.second].push_back(i.first);
        }
        
        for(int i = 0; i < numCourses; i++){
            if(dfs(i, flag, Map))
                return false;
        }
        
        return true;
    }
    
    bool dfs(int num, vector<int>& flag, vector<vector<int>>& Map){
        if(flag[num] == 2) return false;
        if(flag[num] == 1) return true;
        
        // flag[num] = 0
        flag[num] = 1;
        for(auto i : Map[num])
            if(dfs(i, flag, Map))
                return true;
        flag[num] = 2;
        
        return false;
    }
};
```
